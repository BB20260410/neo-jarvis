## 一、Mem0 记忆算法 → Noe 集成方案

### 1.1 单次 ADD-only 提取（替换 FactExtractor）

**Mem0 原理**：一次 LLM 调用完成记忆提取，不执行 UPDATE/DELETE。记忆只增不改。

```javascript
// src/memory/NoeAdditiveExtractor.js — 新建
// 借鉴 Mem0 V3 的 ADD-only 提取

export function createAdditiveExtractor({ llm, memoryStore, embedder }) {
  return async function extract(messages, existingMemories) {
    // Phase 1: 检索现有记忆（top 10）
    const queryEmbed = await embedder.embed(messages);
    const existing = await memoryStore.searchByEmbedding(queryEmbed, { limit: 10 });

    // Phase 2: 反幻觉映射（整数 ID 替换 UUID）
    const uuidMap = {};
    const mapped = existing.map((m, i) => {
      uuidMap[String(i)] = m.id;
      return { id: String(i), text: m.content };
    });

    // Phase 3: 单次 LLM 提取
    const prompt = buildAdditivePrompt(mapped, messages);
    const response = await llm.chat(prompt, { responseFormat: 'json' });
    const extracted = response.memories || [];

    // Phase 4: 哈希去重
    const existingHashes = new Set(existing.map(m => m.hash));
    const seen = new Set();
    const newMemories = [];

    for (const mem of extracted) {
      const hash = md5(mem.text);
      if (existingHashes.has(hash) || seen.has(hash)) continue;
      seen.add(hash);
      newMemories.push({ ...mem, hash });
    }

    // Phase 5: 批量写入
    for (const mem of newMemories) {
      const embedding = await embedder.embed(mem.text);
      memoryStore.write({
        content: mem.text,
        hash: mem.hash,
        embedding,
        salience: 3,
        scope: 'conversation',
      });
    }

    return { added: newMemories.length, total: existing.length + newMemories.length };
  };
}
```

### 1.2 多信号融合检索（增强 recallFused）

**Mem0 原理**：语义 + BM25 关键词 + 实体链接三路融合。

```javascript
// 增强 NoeFusionRanker.js — 加入实体提升

export function hybridScoring(semanticResults, bm25Scores, entityBoosts, topK = 10) {
  const ENTITY_WEIGHT = 0.5;
  const maxPossible = 1.0 + (bm25Scores.size > 0 ? 1.0 : 0) + ENTITY_WEIGHT;

  const scored = semanticResults
    .filter(r => r.score >= 0.3)  // 语义分数门控
    .map(r => {
      const bm25 = bm25Scores.get(r.id) || 0;
      const entity = entityBoosts.get(r.id) || 0;
      const combined = Math.min((r.score + bm25 + entity) / maxPossible, 1.0);
      return { ...r, score: combined };
    })
    .sort((a, b) => b.score - a.score)
    .slice(0, topK);

  return scored;
}
```

### 1.3 BM25 Sigmoid 归一化

```javascript
// Mem0 的 BM25 归一化：根据查询长度调整 sigmoid 参数
function normalizeBM25(rawScore, query) {
  const terms = query.split(/\s+/).length;
  let midpoint, steepness;

  if (terms <= 3)       { midpoint = 5.0;  steepness = 0.7; }
  else if (terms <= 6)  { midpoint = 7.0;  steepness = 0.6; }
  else if (terms <= 9)  { midpoint = 9.0;  steepness = 0.5; }
  else if (terms <= 15) { midpoint = 10.0; steepness = 0.5; }
  else                  { midpoint = 12.0; steepness = 0.5; }

  return 1.0 / (1.0 + Math.exp(-steepness * (rawScore - midpoint)));
}
```

```javascript
// src/memory/NoeBlockMemory.js — 新建
// 借鉴 Letta 的可编辑内存块

export class BlockMemory {
  constructor(blocks = []) {
    this.blocks = new Map();
    blocks.forEach(b => this.blocks.set(b.label, { ...b }));
  }

  // 渲染为系统提示词片段
  compile() {
    let result = '<memory_blocks>\n';
    for (const [label, block] of this.blocks) {
      if (!block.value.trim()) continue;
      result += `<${label}>\n`;
      result += `<description>${block.description || ''}</description>\n`;
      result += `<value>${block.value}</value>\n`;
      result += `</${label}>\n\n`;
    }
    result += '</memory_blocks>';
    return result;
  }

  // Agent 自己调用的工具
  coreMemoryAppend(label, content) {
    const block = this.blocks.get(label);
    if (!block) throw new Error(`Block "${label}" not found`);
    if (block.readOnly) throw new Error(`Block "${label}" is read-only`);
    block.value += '\n' + content;
    block.updatedAt = new Date().toISOString();
  }

  coreMemoryReplace(label, oldContent, newContent) {
    const block = this.blocks.get(label);
    if (!block) throw new Error(`Block "${label}" not found`);
    if (block.readOnly) throw new Error(`Block "${label}" is read-only`);
    if (!block.value.includes(oldContent)) {
      throw new Error(`Content not found in block "${label}"`);
    }
    block.value = block.value.replace(oldContent, newContent);
    block.updatedAt = new Date().toISOString();
  }
}
```

### 2.2 上下文窗口压力管理

```javascript
// 增强 NoeContextBudgeter.js — 加入 Letta 的压力管理

export function createContextComposer({ maxTokens = 6000, llm = null } = {}) {
  const sections = [];

  return {
    add(label, content, keep = 5) {
      sections.push({ label, content, keep, tokens: estimateTokens(content) });
    },

    async compose() {
      const totalTokens = sections.reduce((s, sec) => s + sec.tokens, 0);

      // 压力检测
      if (totalTokens > maxTokens * 0.9 && llm) {
        // 触发自动摘要：保留高 keep 段，摘要低 keep 段
        const lowKeep = sections.filter(s => s.keep <= 3);
        const highKeep = sections.filter(s => s.keep > 3);

        if (lowKeep.length > 0) {
          const summary = await llm.summarize(lowKeep.map(s => s.content).join('\n'));
          sections.length = 0;
          sections.push({ label: 'summary', content: summary, keep: 4, tokens: estimateTokens(summary) });
          highKeep.forEach(s => sections.push(s));
        }
      }

      // 按 keep 升序排列，超预算时丢弃低 keep 段
      sections.sort((a, b) => a.keep - b.keep);
      let budget = maxTokens;
      const kept = [];

      for (const sec of sections) {
        if (budget - sec.tokens >= 0) {
          kept.push(sec);
          budget -= sec.tokens;
        }
      }

      return kept.map(s => s.content).join('\n');
    }
  };
}
```

### 3.2 ActionLoopDetector（防卡死）

```javascript
// 借鉴 browser-use 的循环检测
export class ActionLoopDetector {
  constructor({ windowSize = 20, warnThreshold = 5, criticalThreshold = 8, forceStopThreshold = 12 } = {}) {
    this.history = [];
    this.windowSize = windowSize;
    this.warnThreshold = warnThreshold;
    this.criticalThreshold = criticalThreshold;
    this.forceStopThreshold = forceStopThreshold;
  }

  record(action, pageFingerprint) {
    const hash = md5(JSON.stringify(action) + pageFingerprint);
    this.history.push({ hash, timestamp: Date.now() });
    if (this.history.length > this.windowSize) this.history.shift();

    // 检测重复
    const recent = this.history.slice(-this.windowSize);
    const counts = {};
    recent.forEach(h => { counts[h.hash] = (counts[h.hash] || 0) + 1; });

    const maxRepeat = Math.max(...Object.values(counts));

    if (maxRepeat >= this.forceStopThreshold) {
      return { status: 'force_stop', message: `检测到循环 ${maxRepeat} 次，强制终止` };
    }
    if (maxRepeat >= this.criticalThreshold) {
      return { status: 'critical', message: `检测到循环 ${maxRepeat} 次，强烈建议换策略` };
    }
    if (maxRepeat >= this.warnThreshold) {
      return { status: 'warn', message: `检测到循环 ${maxRepeat} 次` };
    }

    return { status: 'ok' };
  }

  reset() { this.history = []; }
}
```

```javascript
// src/vision/NoeVisualGUIAgent.js — 新建
// 借鉴 UI-TARS 的纯视觉 GUI 操作

export class NoeVisualGUIAgent {
  constructor({ vlmClient, actionExecutor }) {
    this.vlm = vlmClient;
    this.executor = actionExecutor;
    this.COMPUTER_USE_PROMPT = `你是一个 GUI 操作 Agent。根据屏幕截图和任务指令，输出操作。
格式：Thought: 分析屏幕内容
Action: click(start_box='(x, y)') 或 type(text='要输入的文字') 或 scroll(direction='down')`;
  }

  async act(instruction, screenshot) {
    // 1. VLM 理解截图
    const response = await this.vlm.chat({
      messages: [
        { role: 'system', content: this.COMPUTER_USE_PROMPT },
        { role: 'user', content: [
          { type: 'image_url', image_url: screenshot },
          { type: 'text', text: instruction }
        ]}
      ]
    });

    // 2. 解析动作
    const action = this.parseAction(response);

    // 3. 执行
    const result = await this.executor.execute(action);

    return { response, action, result };
  }

  parseAction(response) {
    // 解析 "Action: click(start_box='(500,300)')" 格式
    const clickMatch = response.match(/click\(start_box='?\((\d+),\s*(\d+)\)'?\)/);
    if (clickMatch) {
      return { type: 'click', x: parseInt(clickMatch[1]), y: parseInt(clickMatch[2]) };
    }

    const typeMatch = response.match(/type\(text='(.+?)'\)/);
    if (typeMatch) {
      return { type: 'type', text: typeMatch[1] };
    }

    const scrollMatch = response.match(/scroll\(direction='(\w+)'\)/);
    if (scrollMatch) {
      return { type: 'scroll', direction: scrollMatch[1] };
    }

    return { type: 'unknown', raw: response };
  }
}
```

```javascript
// src/loop/NoeSelfReview.js — 新建
// 借鉴 CowAgent 的 Self-Evolution 机制

export function createSelfReview({ memoryStore, skillStore, commitmentStore, llm, cooldownMs = 600_000 }) {
  let lastReviewAt = 0;
  let lastMessageCount = 0;

  return async function checkAndReview(currentMessageCount) {
    const now = Date.now();

    // 触发条件：空闲超过 10 分钟 + 累计轮次 ≥ 6
    if (now - lastReviewAt < cooldownMs) return null;
    if (currentMessageCount - lastMessageCount < 6) return null;

    lastReviewAt = now;
    lastMessageCount = currentMessageCount;

    // 隔离会话执行审查
    const review = await llm.chat({
      system: `你是 Noe 的自我审查模块。审查最近的对话，找出：
1. 需要巩固的记忆（重要但可能被遗忘的）
2. 可以改进的技能（发现的更好做法）
3. 未完成的任务（承诺但没做的）
输出 JSON: {memories: [...], improvements: [...], pendingTasks: [...]}`,
      messages: [{ role: 'user', content: '审查最近的对话' }]
    });

    // 应用改进
    const result = JSON.parse(review);

    // 巩固记忆
    for (const mem of result.memories || []) {
      memoryStore.bumpSalience(mem.memoryId);
    }

    // 改进技能
    for (const imp of result.improvements || []) {
      skillStore.upsert(imp.skillName, imp.improvedContent);
    }

    // 跟进未完成任务
    for (const task of result.pendingTasks || []) {
      commitmentStore.add({ title: task.title, dueWindow: 'next_interaction' });
    }

    return result;
  };
}
```

## 七、deer-flow Sub-Agent 编排 → Noe 集成方案

### 7.1 Lead + Sub-Agent 架构

```javascript
// src/room/NoeLeadAgent.js — 新建
// 借鉴 deer-flow 的 Sub-Agent 编排

export class NoeLeadAgent {
  constructor({ maxConcurrent = 3, timeoutMs = 900_000, adapters }) {
    this.maxConcurrent = maxConcurrent;
    this.timeoutMs = timeoutMs;
    this.adapters = adapters;
    this.running = new Map();  // taskId → { agent, cancel }
  }

  async decompose(task) {
    // Lead Agent 拆解任务为子任务 DAG
    const plan = await this.adapters.primary.chat({
      system: `将以下任务拆解为可并行执行的子任务。
输出 JSON: {tasks: [{id, description, dependencies, adapter}]}`,
      messages: [{ role: 'user', content: task }]
    });

    return JSON.parse(plan);
  }

  async executePlan(plan) {
    const results = new Map();

    // 拓扑排序执行
    for (const task of this.topologicalSort(plan.tasks)) {
      // 等待依赖完成
      await Promise.all(task.dependencies.map(d => results.get(d)));

      // 并发控制
      while (this.running.size >= this.maxConcurrent) {
        await Promise.race(this.running.values());
      }

      // 启动子 Agent
      const result = await this.runSubAgent(task);
      results.set(task.id, result);
    }

    return results;
  }

  async runSubAgent(task) {
    const controller = new AbortController();
    const timer = setTimeout(() => controller.abort(), this.timeoutMs);

    try {
      const adapter = this.adapters[task.adapter] || this.adapters.primary;
      const result = await adapter.chat({
        messages: [{ role: 'user', content: task.description }],
        signal: controller.signal,
      });
      return { status: 'completed', result };
    } catch (err) {
      return { status: 'failed', error: err.message };
    } finally {
      clearTimeout(timer);
      this.running.delete(task.id);
    }
  }

  topologicalSort(tasks) {
    // 拓扑排序：无依赖的先执行
    const sorted = [];
    const remaining = [...tasks];

    while (remaining.length > 0) {
      const ready = remaining.filter(t =>
        t.dependencies.every(d => sorted.some(s => s.id === d))
      );
      if (ready.length === 0) break;  // 环检测
      sorted.push(...ready);
      remaining.splice(0, ready.length);
    }

    return sorted;
  }
}
```

### 7.2 Middleware 链架构

```javascript
// src/loop/NoeMiddlewareChain.js — 新建
// 借鉴 deer-flow 的 18 层 Middleware

export class NoeMiddlewareChain {
  constructor() {
    this.middlewares = [];
  }

  use(middleware) {
    this.middlewares.push(middleware);
    return this;
  }

  async execute(context) {
    let idx = 0;

    const next = async () => {
      if (idx >= this.middlewares.length) return context;
      const mw = this.middlewares[idx++];
      return mw(context, next);
    };

    return next();
  }
}

// 内置中间件
export const builtInMiddlewares = {
  // 循环检测
  loopDetection: async (ctx, next) => {
    if (ctx.loopDetector?.record(ctx.action)?.status === 'force_stop') {
      ctx.aborted = true;
      ctx.abortReason = 'loop_detected';
      return ctx;
    }
    return next();
  },

  // 预算检查
  budgetCheck: async (ctx, next) => {
    if (ctx.budget?.exceeded) {
      ctx.aborted = true;
      ctx.abortReason = 'budget_exceeded';
      return ctx;
    }
    return next();
  },

  // Token 使用追踪
  tokenUsage: async (ctx, next) => {
    const result = await next();
    if (result.usage) {
      ctx.totalTokens = (ctx.totalTokens || 0) + result.usage.total;
    }
    return result;
  },

  // 记忆提取
  memoryExtract: async (ctx, next) => {
    const result = await next();
    if (result.messages) {
      await ctx.memoryExtractor?.extract(result.messages);
    }
    return result;
  },
```

### 核心发现

1. **Noe 在意识工程层面领先社区**：驱力/目标/内心独白/情感/节律/期望//workspace 8 项能力在 GitHub 上无现成对标项目
2. **Noe 在执行能力上需要补强**：GUI 操作/浏览器自动化/深度研究/进程守护是最大缺口
3. **Noe 在基础设施上可以升级**：记忆衰减/模型速度/向量搜索/VAD 精度都有成熟替代方案
4. **Node.js 语音生态空白**：faster-whisper/Silero VAD/Paraformer 都没有成熟 Node.js 封装，Noe 的 sherpa-onnx-node 已经是领先实现
5. **sqlite-vec 是最大惊喜**：7.7k Stars 的 SQLite 向量搜索扩展，可以替换 Noe 当前的自建 VectorIndex
